9. Palindrome Number

Posted at — Nov 12, 2019
class Solution {
public:
    bool isPalindrome(int x) {
        if (x < 0 || (x % 10 == 0 && x != 0)) return false;
        return x == reverse(x);
    }
    
    int reverse(int x) {
        int res = 0;
        while (x != 0) {
            if (res > INT_MAX / 10) return -1;
            res = res * 10 + x % 10;
            x = x / 10;
        }
        return res;
    }
};
