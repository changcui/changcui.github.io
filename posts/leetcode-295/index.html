Find Median from Data Stream

Posted at — Mar 21, 2020
// 思路一：同剑指 offer
class MedianFinder {
public:
    MedianFinder() {
        cnt = 0;
    }
    
    void addNum(int num) {
        cnt++;
        if (cnt & 1) {
            q.push(num);
            p.push(q.top());
            q.pop();
        } else {
            p.push(num);
            q.push(p.top());
            p.pop();
        }
    }
    
    double findMedian() {
        if (cnt & 1) return p.top();
        return (p.top() + q.top()) / 2.0;
    }
private:
    priority_queue<int, vector<int>, greater<int>> p;
    priority_queue<int, vector<int>> q;
    int cnt;
};
