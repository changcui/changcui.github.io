栈的压入、弹出序列

March 7, 2020

// 思路一：模拟栈的行为，如果压入元素与popV的前列元素相等，则证明 push 之后随即 pop, 然后判断剩下的元素是否后近先出
class Solution {
public:
    bool IsPopOrder(vector<int> pushV,vector<int> popV) {
        stack<int> test;
        int l = 0;
        for (auto i : pushV) {
            if (i == popV[l]) l++;
            else test.push(i);
        }
        while (!test.empty()) {
            if (test.top() != popV[l++]) return false;
            test.pop();
        }
        return true;
    }
};
// 运行时间：4ms
// 占用内存：480k

// 思路二：也是模拟栈的行为，每次 push 的时候，都可以一直 pop，直到栈顶元素与 popV 的前列元素不相等或者栈为空为止
class Solution {
public:
    bool IsPopOrder(vector<int> pushV,vector<int> popV) {
        stack<int> test;
        int l = 0;
        for (auto i : pushV) {
            test.push(i);
            while (!test.empty() && test.top() == popV[l]) {
                test.pop();
                l++;
            }
        }
        return test.empty();
    }
};
// 运行时间：3ms
// 占用内存：468k
