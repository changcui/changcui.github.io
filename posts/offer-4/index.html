重建二叉树

Posted at — Feb 1, 2020
// 思路一：使用递归重建，每次重建一个节点，将两组数在根结点处划分成两段递归调用
// 注意细节处理，第一次调用右节点为 pre.size()-1，调用体的 for loop 使用
// i <= rv - lv，以及出口为 lv > rv
class Solution {
public:
    TreeNode* reConstructBinaryTree(vector<int> pre,vector<int> vin) {
        return helper(pre, vin, 0, pre.size()-1, 0, vin.size()-1);
    }
    TreeNode* helper(vector<int>& pre, vector<int>& vin, 
                     int lp, int rp, int lv, int rv) {
        if (lp > rp || lv > rv) return NULL;
        TreeNode* rtn = new TreeNode(pre[lp]);
        if (lp == rp && lv == rv) return rtn;
        for (int i = 0; i <= rv - lv; i++) {
            if (vin[i+lv] == pre[lp]) {
                rtn -> left = helper(pre, vin, lp+1, lp+i, lv, lv+i-1);
                rtn -> right = helper(pre, vin, lp+i+1, rp, lv+i+1, rv);
                break;
            }
        }    
        return rtn;
    }
};
// 运行时间：3ms
// 占用内存：480k
