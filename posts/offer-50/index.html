数组中重复的数字

Posted at — Mar 12, 2020
// 思路一：使用 set 保存数字
class Solution {
public:
    bool duplicate(int numbers[], int length, int* duplication) {
        unordered_set<int> s;
        for (int i = 0; i < length; i++) {
            if (s.count(numbers[i])) {
                *duplication = numbers[i];
                return true;
            }
            s.insert(numbers[i]);
        }
        return false;
    }
};
// 运行时间：4ms
// 占用内存：492k

// 思路二：每个元素都应该在它下标的位置
class Solution {
public:
    bool duplicate(int numbers[], int length, int* duplication) {
        for (int i = 0; i < length; i++) {
            while (numbers[i] != i) {
                if (numbers[i] == numbers[numbers[i]]) {
                    *duplication = numbers[i];
                    return true;
                }
                swap(numbers[i], numbers[numbers[i]]);
            }
        }
        return false;
    }
};


// 运行时间：5ms
// 占用内存：540k
