1025. Divisor Game

Posted at — Aug 31, 2019

dp

class Solution {
public:
    bool divisorGame(int N) {
        if (N == 1)
            return false;
        if (N == 2)
            return true;
        vector<bool> saved;
        saved.push_back(false);
        saved.push_back(true);
        for (int i = 3; i <= N; i++) {
            bool res = false;
            for (int j = 1; j * j < i; j++) {
                if (i % j == 0 && !saved[i - j - 1]) {
                    res = true;
                    break;
                }
            }
            saved.push_back(res);
        }
        return saved.back();
    }
};
