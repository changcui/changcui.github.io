左旋转字符串

Posted at — Mar 11, 2020
// 思路一：左删除右添加
class Solution {
public:
    string LeftRotateString(string str, int n) {
        if (str.size() == 0) return str;
        n = n % str.size();
        while (n--) {
            char a = str[0];
            str.erase(str.begin());
            str += a;
        }
        return str;
    }
};
// 运行时间：3ms
// 占用内存：480k

// 思路二：先反转字符串，然后反转前半部分，再反转后半部分，(AB)T = BTAT
class Solution {
public:
    string LeftRotateString(string str, int n) {
        if (str.size() == 0) return str;
        n = n % str.size();
        reverse(str.begin(), str.end());
        reverse(str.begin(), str.begin() + str.size() - n);
        reverse(str.begin() + str.size() - n, str.end());
        return str;
    }
};
// 运行时间：2ms
// 占用内存：476k
