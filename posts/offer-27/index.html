字符串的排列

Posted at — Mar 8, 2020
// 思路一：列出所有结果，考虑 DFS，出口是 size 一致，注意重复的剪枝
class Solution {
public:
    vector<string> Permutation(string str) {
        vector<string> result;
        if (str.size() == 0) return result;
        vector<bool> used(str.size(), false);
        sort(str.begin(), str.end());
        dfs(str, "", used, result);
        return result;
    }
    
    void dfs(string& str, string out, vector<bool>& used, 
             vector<string>& result) {
        if (str.size() == out.size()) {result.push_back(out); return;}
        for (int i = 0; i < str.size(); i++) {
            if (i > 0 && str[i] == str[i-1] && !used[i-1]) continue;
            if (!used[i]) {
                used[i] = true; 
                dfs(str, out+str[i], used, result);
                used[i] = false;
            }
        }
    }
};
// 运行时间：3ms
// 占用内存：488k

class Solution {
public:
    vector<string> Permutation(string str) {
        vector<string> result;
        if (str.size() == 0) return result;
        sort(str.begin(), str.end());
        dfs(str, "", result);
        return result;
    }
    
    void dfs(string& str, string out, vector<string>& result) {
        if (str.size() == 0) {result.push_back(out); return;}
        for (int i = 0; i < str.size(); i++) {
            if (i > 0 && str[i] == str[i-1]) continue;
            auto c = str[i];
            str.erase(str.begin()+i);
            dfs(str, out+c, result);
            str.insert(str.begin()+i, c);
        }
    }
};
// 运行时间：3ms
// 占用内存：464k
