684. Redundant Connection

Posted at — Sep 21, 2019
class Solution {
public:
    vector<int> fa;
    int find(int a) {
        return a == fa[a] ? a : a = find(fa[a]);
    }
    vector<int> findRedundantConnection(vector<vector<int>>& edges) {
        auto N = edges.size();
        fa.resize(N + 1);
        for (auto i = 1; i < N + 1; i++) {
            fa[i] = i;
        }
        int l, r;
        for (auto e : edges) {
            auto i_f = find(e[0]);
            auto j_f = find(e[1]);
            if (i_f != j_f) {
                fa[i_f] = j_f;
            } else {
                l = e[0];
                r = e[1];
            }
        }
        return vector<int>{l, r};
    }
};
