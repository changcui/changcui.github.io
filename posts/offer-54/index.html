字符流中第一个不重复的字符

Posted at — Mar 12, 2020
// 思路一：储存初次出现的下标，遍历整个数组，寻找最小下标，空间复杂度O(k)，时间复杂度O(k)
class Solution
{
public:
    vector<int> c;
    int cnt;
    void Insert(char ch) {
        if (c.empty()) c.resize(256, INT_MAX), cnt = 0;
        c[ch] = c[ch] == INT_MAX ? cnt : -1;
        cnt++;
    }
  //return the first appearence once char in current stringstream
    char FirstAppearingOnce() {
        char rtn = '#';
        int idx = INT_MAX;
        for (int i = 0; i < 256; i++) {
            if (c[i] != -1 && c[i] < idx) rtn = i, idx = c[i];
        }
        return rtn;
    }

};
// 运行时间：3ms
// 占用内存：488k
