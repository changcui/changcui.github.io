表示数值的字符串

Posted at — Mar 12, 2020
// 思路一：拆分成三部分去 check
class Solution {
public:
    bool isInt(string s) {
        if (s.empty()) return false;
        if (s.size() == 1 && (s[0] == '+' || s[0] == '-')) return true;
        if (!isdigit(s[0]) && s[0] != '+' && s[0] != '-') return false;
        for (auto i = 1; i < s.size(); i++) {
            if (!isdigit(s[i])) return false;
        }
        return true;
    }
    bool isFloat(string s) {
        auto p = s.find('.');
        if (p != std::string::npos) {
            return isInt(s.substr(0, p)) && isInt(s.substr(p+1));
        }
        return isInt(s);
    }
    bool isNumeric(char* strin) {
        string s(strin);
        auto p = s.find_first_of("eE");
        if (p != std::string::npos) {
            return isFloat(s.substr(0, p)) && isInt(s.substr(p+1));
        }
        return isFloat(s);
    }
};
// 运行时间：5ms
// 占用内存：512k
