从上往下打印二叉树

March 7, 2020

// 思路一：层序遍历，用 queue 保存当前层，遍历当前层，加入下一层
class Solution {
public:
    vector<int> PrintFromTopToBottom(TreeNode* root) {
        vector<int> result;
        if (!root) return result;
        queue<TreeNode*> q;
        q.push(root);
        while (!q.empty()) {
            int c = q.size();
            while (c--) {
                auto t = q.front(); q.pop();
                result.push_back(t->val);
                if (t->left) q.push(t->left);
                if (t->right) q.push(t->right);
            }
        }
        return result;
    }
};
// 运行时间：3ms
// 占用内存：604k

// 思路二：因为结果不需要区分层一层，可以再优化一下，去掉内层循环，但复杂度一致
class Solution {
public:
    vector<int> PrintFromTopToBottom(TreeNode* root) {
        vector<int> result;
        if (!root) return result;
        queue<TreeNode*> q;
        q.push(root);
        while (!q.empty()) {
            auto t = q.front(); q.pop();
            result.push_back(t->val);
            if (t->left) q.push(t->left);
            if (t->right) q.push(t->right);
        }
        return result;
    }
};
// 运行时间：4ms
// 占用内存：484k
