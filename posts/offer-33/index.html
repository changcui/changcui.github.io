//  思路一：分三组累乘，取最小的，并更新下标
class Solution {
public:
    int GetUglyNumber_Solution(int index) {
        if (index == 0) return 0;
        vector<int> nums{1};
        int i1 = 0, i2 = 0, i3 = 0;
        while (--index) {
            int n1 = nums[i1] * 2;
            int n2 = nums[i2] * 3;
            int n3 = nums[i3] * 5;
            int min_ = min(n1, min(n2, n3));
            if (min_ == n1) i1++;
            if (min_ == n2) i2++;
            if (min_ == n3) i3++;
            nums.push_back(min_);
        }
        return nums.back();
    }
};
// 运行时间：3ms
// 占用内存：612k
