Serialize and Deserialize Binary Tree

Posted at — Mar 21, 2020
// 思路一：同剑指 offer
class Codec {
public:

    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        string result = "";
        sePre(root, result);
        return result;
    }
    
    void sePre(TreeNode* root, string& result) {
        if (!root) {result += "#!"; return;}
        result += to_string(root->val) + "!";
        sePre(root->left, result);
        sePre(root->right, result);
    }
    
    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        return desePre(data);
    }
    
    TreeNode* desePre(string& data) {
        if (data.size() == 0) return NULL;
        if (data[0] == '#') {
            data = data.substr(2);
            return NULL;
        }
        TreeNode* root = new TreeNode(stoi(data.substr(0, data.find('!'))));
        data = data.substr(data.find('!') + 1);
        root->left = desePre(data);
        root->right = desePre(data);
        return root;
    }
};
