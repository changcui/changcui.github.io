斐波那契数列

Posted at — Feb 1, 2020
// 思路一：递归如果不储存的话会超时，使用循环
class Solution {
public:
    int Fibonacci(int n) {
        int a = 0, b = 1;
        while (n--) {
            int tmp = a + b;
            a = b;
            b = tmp;
        }
        return a;
    }
};
// 运行时间：4ms
// 占用内存：484k
