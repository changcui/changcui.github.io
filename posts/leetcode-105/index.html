105.Construct Binary Tree from Preorder and Inorder Traversal

Posted at — Dec 13, 2019
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        int n = preorder.size() - 1;
        return helper(preorder, inorder, 0, n, 0, n);
    }
    TreeNode* helper(vector<int>& preorder, vector<int>& inorder, int lp, int rp,
                     int li, int ri) {
        if (lp > rp || li > ri) return nullptr;
        if (lp == rp && li == ri) {
            auto tmp = new TreeNode(inorder[li]);
            return tmp;
        }
        for (int i = li; i <= ri; i++) {
            if (inorder[i] == preorder[lp]) {
                auto tmp = new TreeNode(inorder[i]);
                auto off = i - li;
                tmp -> left = helper(preorder, inorder, lp + 1, lp + off, li, i - 1);
                tmp -> right = helper(preorder, inorder, lp + off + 1, rp, i + 1, ri);
                return tmp;
            }
        }
        return nullptr;
    }
};
