190. Reverse Bits

Posted at — Sep 4, 2019
class Solution {
public:
    uint32_t reverseBits(uint32_t n) {
        uint32_t rtn = 0;
        for (int i = 0; i < 32; i++) {
            rtn <<= 1;
            rtn |= (n >> i) & 1;
        }
        return rtn;
    }
};
