1160. Find Words That Can Be Formed by Characters

Posted at — Aug 30, 2019

vector faster than map

class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        vector<int> cnt(255, 0);
        for (auto c : chars) {
            cnt[c] += 1;
        }
        auto rtn = 0;
        for (auto word : words) {
            auto tmp = cnt;
            auto flag = true;
            for (auto c : word) {
                if (tmp[c] == 0) {
                    flag = false;
                    break;
                }
                else
                    tmp[c] -= 1;
            }
            rtn += flag ? word.size() : 0;
        }
        return rtn;
    }
};
