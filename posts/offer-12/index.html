数值的整数次方

Posted at — Feb 5, 2020
// 思路一：直接使用内置函数
class Solution {
public:
    double Power(double base, int exponent) {
        return pow(base, exponent);
    }
};
// 运行时间：5ms
// 占用内存：508k

// 思路二：一个一个乘，需要判断正负
class Solution {
public:
    double Power(double base, int exponent) {
        bool flag = exponent > 0 ? true : false;
        exponent = exponent > 0 ? exponent : -exponent;
        double rtn = 1.0;
        while (exponent--) rtn *= base;
        return flag ? rtn : 1 / rtn;
    }
};
// 运行时间：5ms
// 占用内存：476k

// 思路三：使用二分递归，需要根据递归判断返回值
class Solution {
public:
    double Power(double base, int exponent) {
        bool flag = exponent > 0 ? true : false;
        exponent = exponent > 0 ? exponent : -exponent;
        double rtn = helper(base, exponent);
        return flag ? rtn : 1 / rtn;
    }
    
    double helper(double base, int exponent) {
        if (exponent == 0) return 1.0;
        if (exponent == 1) return base;
        double sub = helper(base, exponent/2);
        return exponent & 1 ? base * sub * sub : sub * sub;
    }
};
// 运行时间：4ms
// 占用内存：464k

// 思路四：快速幂，通过判断 exp 的每一位来进行累乘，复杂度为O(1)
class Solution {
public:
    double Power(double base, int exponent) {
        bool flag = exponent > 0 ? true : false;
        exponent = exponent > 0 ? exponent : -exponent;
        double rtn = 1.0;
        while (exponent) {
            if (exponent & 1) rtn *= base;
            base *= base;
            exponent >>= 1;
        }
        return flag ? rtn : 1 / rtn;
    }
};
// 运行时间：4ms
// 占用内存：632k
