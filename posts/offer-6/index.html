旋转数组的最小数字

Posted at — Feb 1, 2020
// 思路一：二分查找，如果左值小于右值，必定是递增，返回左值；如果中间值大于左值
// 则必在右边，若在左边则左值大于右值大于中间值，同理，如果中间值小于右值，必在
// 左边，若在右边则中间值大于左值大于右值，else 的情况只能为 l++，不能为 r--
class Solution {
public:
    int minNumberInRotateArray(vector<int> rotateArray) {
        if (rotateArray.empty()) return 0;
        int l = 0, r = rotateArray.size() - 1;
        while (l < r) {
            if (rotateArray[l] < rotateArray[r]) return rotateArray[l];
            int mid = l + (r - l) / 2;
            if (rotateArray[mid] > rotateArray[l]) l = mid + 1;
            else if (rotateArray[mid] < rotateArray[r]) r = mid;
            else l++;
        }
        return rotateArray[l];
    }
};
// 运行时间：27ms
// 占用内存：612k

// 思路二：也是二分，判断条件变化了一些，更容易理解，只与右边的比较，如果大的话，肯定 
// 在右边，因为有一个减少过程，如果小的话，肯定在左边，与上文相同，如果相等的话，无法
// 判断，让 r--。
class Solution {
public:
    int minNumberInRotateArray(vector<int> rotateArray) {
        if (rotateArray.empty()) return 0;
        int l = 0, r = rotateArray.size() - 1;
        while (l < r) {
            if (rotateArray[l] < rotateArray[r]) return rotateArray[l];
            int mid = l + (r - l) / 2;
            if (rotateArray[mid] > rotateArray[r]) l = mid + 1;
            else if (rotateArray[mid] < rotateArray[r]) r = mid;
            else r--;
        }
        return rotateArray[l];
    }
};
// 运行时间：32ms
// 占用内存：616k
