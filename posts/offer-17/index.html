树的子结构

Posted at — Feb 26, 2020
// 思路一：树的问题大多要用递归，当根结点值相等时，需要判断是否是找的子结构
class Solution {
public:
    bool IsSubtree(TreeNode* a, TreeNode* b) {
        if (!b) return true;
        if (!a) return false;
        return a->val == b->val && IsSubtree(a->left, b->left) &&
            IsSubtree(a->right, b->right);
    }
    bool HasSubtree(TreeNode* pRoot1, TreeNode* pRoot2) {
        if (!pRoot2 || !pRoot1) return false;
        return (pRoot1->val == pRoot2->val && IsSubtree(pRoot1, pRoot2)) || 
            HasSubtree(pRoot1->left, pRoot2) || 
            HasSubtree(pRoot1->right, pRoot2);
    }
};
// 运行时间：3ms
// 占用内存：376k
