正则表达式匹配

March 12, 2020

// 思路一：按照 pattern 下一位是不是 * 来分情况考虑，如果是 * 的话，可以使用，也可以不使用，分两种情况返回。
class Solution {
public:
    int sl, pl;
    bool match(char* str, char* pattern) {
        sl = strlen(str), pl = strlen(pattern);
        return helper(str, pattern, 0, 0);
    }
    bool helper(char* str, char* pattern, int i, int j) {
        if (i == sl && j == pl) return true;
        if (j == pl) return false;
        if (j + 1 < pl && pattern[j+1] == '*') {
            if (i < sl && (str[i] == pattern[j] || pattern[j] == '.')) {
                return helper(str, pattern, i + 1, j) ||
                       helper(str, pattern, i, j + 2);
            }
            return helper(str, pattern, i, j + 2);
        }
        if (str[i] == pattern[j] || pattern[j] == '.')
            return helper(str, pattern, i + 1, j + 1);
        return false;
    }
};
// 运行时间：2ms
// 占用内存：364k
