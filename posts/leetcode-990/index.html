990. Satisfiability of Equality Equations

Posted at — Sep 21, 2019
class Solution {
public:
    vector<int> fa;
    int find(int x) {
        return x == fa[x] ? x : x = find(fa[x]); 
    }
    bool equationsPossible(vector<string>& equations) {
        fa.resize(255);
        for (int i = 'a'; i <= 'z'; i++)
            fa[i] = i;
        for (auto e : equations) {
            auto f = e[0];
            auto l = e[3];
            if (e[1] == '=') 
                fa[find(f)] = find(l);
        }
        for (auto e : equations) {
            if (e[1] == '!' && find(e[0]) == find(e[3]))
                return false;     
        }
        return true;
    }
};
