矩形覆盖

Posted at — Feb 3, 2020
// 思路一：和斐波那契一样，递归会超时
class Solution {
public:
    int rectCover(int number) {
        if (number == 0) return 0;
        int a = 1, b = 2;
        while (--number) {
            int tmp = b;
            b = a + b;
            a = tmp;
        }
        return a;
    }
};
// 运行时间：4ms
// 占用内存：464k
