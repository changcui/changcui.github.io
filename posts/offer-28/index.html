数组中出现次数超过一半的数字

Posted at — Mar 8, 2020
// 思路一：使用 map 保存出现的次数，大于一半就返回
class Solution {
public:
    int MoreThanHalfNum_Solution(vector<int> numbers) {
        int n = numbers.size() / 2 + 1;
        unordered_map<int, int> m;
        for (auto i : numbers) {
            m[i] += 1;
            if (m[i] >= n) return i;
        }
        return 0;
    }
};
// 运行时间：3ms
// 占用内存：484k
