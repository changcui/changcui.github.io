394. Decode String

Posted at — Oct 12, 2019
class Solution {
public:
    string decodeString(string s) {
        string rtn = "";
        for (int i = 0; i < s.size();) {
            if (isdigit(s[i])) {
                auto start = i;
                auto end = i;
                while (isdigit(s[++end]));
                auto n = stoi(s.substr(start, end-start+1));
                start = end + 1;
                auto cnt = 1;
                while (cnt != 0) {
                    end += 1;
                    if (s[end] == '[') ++cnt;
                    else if (s[end] == ']') --cnt;
                }
                auto con = decodeString(s.substr(start, end-start));
                for (int j = 0; j < n; j++) rtn += con;
                i = end + 1;
            } else {
                rtn += s[i];
                i++;
            }
        }
        return rtn;
    }
};
