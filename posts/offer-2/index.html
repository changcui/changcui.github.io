替换空格

Posted at — Feb 1, 2020
// 思路一：c++ 处理这种比较麻烦，python 或者 java 应该很简单，思路是先计算替换后
// 的长度然后从后往前替换
class Solution {
public:
    void replaceSpace(char *str,int length) {
        int newL = strlen(str), oldL = strlen(str);
        for (int i = 0; i < oldL; i++) if (str[i] == ' ') newL += 2;
        if (newL >= length) return;
        int idx = newL;
        for (int i = oldL; i >= 0; i--) {
            if (str[i] == ' ') {
                str[idx--] = '0';
                str[idx--] = '2';
                str[idx--] = '%';
            } else str[idx--] = str[i];
        }
    }
};
// 运行时间：3ms
// 占用内存：476k
