75. Sort Colors

Posted at — Sep 8, 2019
class Solution {
public:
    void sortColors(vector<int>& nums) {
        if (nums.size() == 0) {
            return;
        }
        auto len = nums.size();
        for (int i = 0; i < len;) {
            if (nums[i] == 0) {
                nums.insert(nums.begin(), 0);
                nums.erase(nums.begin() + i + 1);
                i += 1;
            } else if (nums[i] == 2) {
                nums.push_back(2);
                nums.erase(nums.begin() + i);
                int j = i;
                while (j < len && nums[j] == 2) {
                    j++;
                }
                if (j == len) {
                    break;
                }
            } else {
                i += 1;
            }
        }
    }
};
