二叉搜索树与双向链表

Posted at — Mar 8, 2020
// 思路一：因为是排序的双向链表，所以考虑中序遍历加递归，在没有连接之间的节点建立连接即可
class Solution {
public:
    TreeNode* Convert(TreeNode* pRootOfTree) {
        if (!pRootOfTree) return NULL;
        
        auto l = Convert(pRootOfTree->left);
        auto t = l;
        while (t && t->right) t = t->right;
        if (t) t->right = pRootOfTree;
        pRootOfTree->left = t;
        
        auto r = Convert(pRootOfTree->right);
        if (r) r->left = pRootOfTree;
        pRootOfTree->right = r;
        return l ? l : pRootOfTree;
    }
};
// 运行时间：3ms
// 占用内存：616k

// 思路二：维护一个前驱指针，使用中序遍历
class Solution {
public:
    TreeNode* Convert(TreeNode* pRootOfTree) {
        if (!pRootOfTree) return NULL;
        TreeNode* pre = NULL;
        helper(pRootOfTree, pre);
        auto rtn = pRootOfTree;
        while (rtn->left) rtn = rtn->left;
        return rtn;
    }
    void helper(TreeNode* root, TreeNode*& pre) {
        if (!root) return;
        helper(root->left, pre);
        root->left = pre;
        if (pre) pre->right = root;
        pre = root;
        helper(root->right, pre);
    }
};
// 运行时间：3ms
// 占用内存：472k
