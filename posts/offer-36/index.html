两个链表的第一个公共结点

Posted at — Mar 9, 2020
class Solution {
public:
    ListNode* FindFirstCommonNode(ListNode* pHead1, ListNode* pHead2) {
        auto p = pHead1, q = pHead2;
        while (p || q) {
            if (p == q) return p;
            if (!p) p = pHead2, q = q->next;
            else if (!q) q = pHead1, p = p->next;
            else p = p->next, q = q->next;
        }
        return NULL;
    }
};
// 运行时间：3ms
// 占用内存：484k
