跳台阶

Posted at — Feb 1, 2020
// 思路一： 使用递归非常简单，时间较长，因为有大量的重复计算
class Solution {
public:
    int jumpFloor(int number) {
        if (number == 1 || number == 0) return 1;
        return jumpFloor(number - 1) + jumpFloor(number - 2);
    }
};
// 运行时间：576ms
// 占用内存：488k

// 思路二： 使用循环，和斐波那契数列基本上一样
class Solution {
public:
    int jumpFloor(int number) {
        int a = 1, b = 1;
        while (number--) {
            int tmp = a + b;
            a = b;
            b = tmp;
        }
        return a;
    }
};
// 运行时间：3ms
// 占用内存：460k
