二进制中1的个数

Posted at — Feb 3, 2020
// 思路一：n = n&(n-1) 总会把最右边的1变为0
class Solution {
public:
     int  NumberOf1(int n) {
         int cnt = 0;
         while (n) {cnt++; n = n&(n-1);}
         return cnt;
     }
};
// 运行时间：3ms
// 占用内存：472k

// 思路二：一位一位的判断是否为1
class Solution {
public:
     int  NumberOf1(int n) {
         int cnt = 0;
         for (int i = 0; i < 32; i++) {
             cnt += (n >> i) & 1;
         }
         return cnt;
     }
};
// 运行时间：5ms
// 占用内存：512k
