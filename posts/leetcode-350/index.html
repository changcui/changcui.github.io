350. Intersection of Two Arrays II

Posted at — Aug 31, 2019
class Solution {
public:
    vector<int> intersect(vector<int>& nums1, vector<int>& nums2) {
        map<int, int> cnt;
        for (auto num: nums1) {
            cnt[num] += 1;
        }
        vector<int> rtn;
        for (auto num: nums2) {
            auto it = cnt.find(num);
            if (it == cnt.end() || it -> second == 0)
                continue;
            rtn.push_back(num);
            it -> second -= 1;
        }
        return rtn;
    }
};

class Solution {
public:
    vector<int> intersect(vector<int>& nums1, vector<int>& nums2) {
        multiset<int> n1(nums1.begin(), nums1.end()), res;
        for (auto n : nums2) {
            auto it = n1.find(n);
            if (it != n1.end()) {
                res.insert(n);
                n1.erase(it);
            }
        }
        return vector<int>(res.begin(), res.end());
    }
};
