1103. Distribute Candies to People

Posted at — Aug 28, 2019

math

class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        int epoch = 0;
        for (; ; epoch++) {
            int needed = (epoch + 1) * num_people * ((epoch + 1) * num_people + 1) / 2;
            if (candies <= needed)
                break;
        }
        vector<int> result(num_people, 0);
        for (int i = 1; i <= num_people; i++) {
            result[i - 1] += epoch * i + num_people * epoch * (epoch - 1) / 2;
        }
        candies -= epoch * num_people * (epoch * num_people + 1) / 2;
        for (int i = 1; i <= num_people; i++) {
            int needed = i + epoch * num_people;
            if (candies <= needed) {
                result[i - 1] += candies;
                break;
            } else {
                result[i - 1] += needed;
                candies -= needed;
            }
        }
        return result;
    }
};
