56. Merge Intervals

Posted at — Sep 21, 2019

greedy

class Solution {
public:
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        vector<vector<int>> result;
        if (intervals.size() == 0)
            return result;
        sort(intervals.begin(), intervals.end(), [](auto i, auto j) {return i.front() < j.front();});
        auto left = intervals[0][0];
        auto right = intervals[0][1];
        for (int i = 1; i < intervals.size(); i++) {
            if (intervals[i][0] <= right) {
                right = max(right, intervals[i][1]);
                continue;
            }
            result.push_back(vector<int>{left, right});
            left = intervals[i][0];
            right = intervals[i][1];
        }
        result.push_back(vector<int>{left, right});
        return result;
    }
};
