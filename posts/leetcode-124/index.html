124.Binary Tree Maximum Path Sum

Posted at — Jun 30, 2020
// 思路一：maxPathDown 从 input node 开始最大的 path 值（左右节点只能选一个），Sum 则为左右节点都选，选左，选右，不选四种情况的最大值
class Solution {
public:
    int maxPathSum(TreeNode* root) {
        int max_ = INT_MIN;
        maxPathDown(root, max_);
        return max_;
    }
    
    int maxPathDown(TreeNode* root, int& max_) {
        if (!root) return 0;
        auto l = max(0, maxPathDown(root->left, max_));
        auto r = max(0, maxPathDown(root->right, max_));
        max_ = max(max_, root->val + l + r);
        return max(l, r) + root->val;
    }
};
