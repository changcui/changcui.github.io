二叉树中和为某一值的路径

Posted at — Mar 7, 2020
// 思路一：dfs，出口是叶子节点且 target 为根结点的值
class Solution {
public:
    vector<vector<int> > FindPath(TreeNode* root,int expectNumber) {
        vector<vector<int> > result;
        if (!root) return result;
        dfs(root, expectNumber, {}, result);
        sort(result.begin(), result.end(), [](auto a, auto b) {
            return a.size() > b.size();});
        return result;
    }
    
    void dfs(TreeNode* root, int target, vector<int> out, 
        vector<vector<int> >& result) {
        out.push_back(root->val);
        if (target == root->val && !root->left && !root->right) {
            result.push_back(out);
            return;
        }
        if (root->left) dfs(root->left, target-root->val, out, result);
        if (root->right) dfs(root->right, target-root->val, out, result);
    }
};
// 运行时间：3ms
// 占用内存：480k
