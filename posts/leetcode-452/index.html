452. Minimum Number of Arrows to Burst Balloons

Posted at — Sep 21, 2019

greedy

class Solution {
public:
    int findMinArrowShots(vector<vector<int>>& points) {
        auto len = points.size();
        if (len == 0) {
            return 0;
        }
        sort(points.begin(), points.end(), [](auto p1, auto p2) {return p1.back() < p2.back();});
        auto cur = points[0][1];
        auto ans = 1;
        for (int i = 1; i < len; i++) {
            if (points[i][0] <= cur) {
                continue;
            }
            ans += 1;
            cur = points[i][1];
        }
        return ans;
    }
};
