Kth Largest Element in an Array

Posted at — Mar 21, 2020
// 思路一：使用 priority_queue 保存最大的 k 的元素
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        priority_queue<int, vector<int>, greater<int>> q;
        for (auto i : nums) {
            if (q.size() < k) q.push(i);
            else if (i > q.top()) {
                    q.push(i);
                    q.pop();
                }
        }
        return q.top();
    }
};

// 思路二：使用快排，根据中枢点在数组中的位置与 k 对比
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        int l = 0, r = nums.size() - 1;
        while (true) {
            int pos = partition(nums, l, r);
            if (pos == k - 1) return nums[k-1];
            else if (pos > k - 1) r = pos - 1;
            else l = pos + 1;
        }
        return -1;
    }
    
    int partition(vector<int>& nums, int l, int r) {
        int pivot = nums[l], i = l + 1, j = r;
        while (i <= j) {
            if (nums[i] < pivot && nums[j] > pivot) swap(nums[i++], nums[j--]);
            if (nums[i] >= pivot) i++;
            if (nums[j] <= pivot) j--;
        }
        swap(nums[l], nums[j]);
        return j;
    }
};
