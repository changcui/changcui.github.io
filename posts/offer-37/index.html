数字在排序数组中出现的次数

Posted at — Mar 10, 2020
// 思路一：直接遍历查找
class Solution {
public:
    int GetNumberOfK(vector<int> data ,int k) {
        int cnt = 0;
        for (auto i : data) cnt += k == i;
        return cnt;
    }
};
// 运行时间：3ms
// 占用内存：484k

// 思路二：二分查找
class Solution {
public:
    int GetNumberOfK(vector<int> data ,int k) {
        int cnt = 0;
        int l = 0, r = data.size() - 1;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (k == data[mid]) {
                int i = mid;
                while (data[i++] == k) cnt++;
                i = mid - 1;
                while (data[i--] == k) cnt++;
                return cnt;
            }
            else if (k > data[mid]) l = mid + 1;
            else r = mid - 1;
        }
        return cnt;
    }
};
// 运行时间：3ms
// 占用内存：492k

// 思路三：STL 大法
class Solution {
public:
    int GetNumberOfK(vector<int> data ,int k) {
        auto range = equal_range(data.begin(), data.end(), k);
        return range.second - range.first;
    }
};
// 运行时间：3ms
// 占用内存：496k
