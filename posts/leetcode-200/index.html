Number of Islands

Posted at — Oct 18, 2019
// 思路一：遇到1，就把临近的1变为0，并计数
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        if (grid.size() == 0) return 0;
        int cnt = 0, m = grid.size(), n = grid[0].size();
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == '1') {
                    helper(grid, i, j);
                    cnt++;
                }
            }
        }
        return cnt;
    }

    void helper(vector<vector<char>>& grid, int i, int j) {
        int m = grid.size(), n = grid[0].size();
        if (i < 0 || j < 0 || i >= m || j >= n || grid[i][j] != '1') return;
        grid[i][j] = '0';
        helper(grid, i+1, j);
        helper(grid, i-1, j);
        helper(grid, i, j+1);
        helper(grid, i, j-1);
    }
};

// 思路二：BFS 非递归
class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        if (grid.size() == 0) return 0;
        int cnt = 0, m = grid.size(), n = grid[0].size();
        vector<int> dirX{0, 1, 0, -1}, dirY{1, 0, -1, 0};
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == '1') {
                    cnt++;
                    queue<pair<int, int>> q{{make_pair(i, j)}};
                    grid[i][j] = '0';
                    while (!q.empty()) {
                        auto t = q.front(); q.pop();
                        for (int k = 0; k < 4; k++) {
                            int x = t.first + dirX[k], y = t.second + dirY[k];
                            if (x < m && x >= 0 && y < n && y >= 0 && 
                                grid[x][y] == '1') {
                                q.push(make_pair(x, y));
                                grid[x][y] = '0';
                            }
                        }
                    }
                }
            }
        }
        return cnt;
    }
};
