剪绳子

Posted at — Mar 19, 2020
// 思路一：贪心，优先剪3
class Solution {
public:
    int cutRope(int number) {
        if (number == 2 || number == 3) return number - 1;
        if (number % 3 == 0) return pow(3, number/3);
        if (number % 3 == 1) return pow(3, number/3-1) * 4;
        if (number % 3 == 2) return pow(3, number/3) * 2;
    }
};
// 运行时间：5ms
// 占用内存：504k
