13. Roman to Integer

Posted at — Sep 6, 2019
class Solution {
public:
    int romanToInt(string s) {
        auto rtn = 0;
        auto l = s.size();
        for (int i = 0; i < l;) {
            if (s[i] == 'I') {
                if (i + 1 < l && s[i + 1] == 'V') {
                    rtn += 4;
                    i += 1;
                } else if (i + 1 < l && s[i + 1] == 'X') {
                    rtn += 9;
                    i += 1;
                } else {
                    rtn += 1;
                }
            }
            else if (s[i] == 'X') {
                if (i + 1 < l && s[i + 1] == 'L') {
                    rtn += 40;
                    i += 1;
                } else if (i + 1 < l && s[i + 1] == 'C') {
                    rtn += 90;
                    i += 1;
                } else {
                    rtn += 10;
                }
            }
            else if (s[i] == 'C') {
                if (i + 1 < l && s[i + 1] == 'D') {
                    rtn += 400;
                    i += 1;
                } else if (i + 1 < l && s[i + 1] == 'M') {
                    rtn += 900;
                    i += 1;
                } else {
                    rtn += 100;
                }
            }
            else if (s[i] == 'V') {
                rtn += 5;
            }
            else if (s[i] == 'L') {
                rtn += 50;
            }
            else if (s[i] == 'D') {
                rtn += 500;
            }
            else {
                rtn += 1000;
            }
            i += 1;
        }
        return rtn;
    }
};
