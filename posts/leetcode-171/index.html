171. Excel Sheet Column Number

Posted at — Sep 6, 2019

long long

class Solution {
public:
    int titleToNumber(string s) {
        long long rtn = 0;
        for (auto i : s) {
            rtn = 26 *  rtn + i - 'A' + 1;
        }
        return rtn;
    }
};
