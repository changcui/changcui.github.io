1051. Height Checker

Posted at — Aug 26, 2019

sort

class Solution {
public:
    int heightChecker(vector<int>& heights) {
        auto sorted = heights;
        sort(sorted.begin(), sorted.end());
        auto result = 0;
        for (int i = 0; i < heights.size(); i++) {
            if (sorted[i] != heights[i]) {
                result++;
            }
        }
        return result;
    }
};
