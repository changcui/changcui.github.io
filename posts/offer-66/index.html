机器人的运动范围

Posted at — Mar 19, 2020
// 思路一：DFS
class Solution {
public:
    int sum_(int r, int l) {
        int rtn = 0;
        while (r || l) {
            rtn += r % 10 + l % 10;
            r /= 10;
            l /= 10;
        }
        return rtn;
    }
    int movingCount(int threshold, int rows, int cols) {
        int rtn = 0;
        vector<vector<bool> > visited(rows, vector<bool>(cols, false));
        helper(rtn, threshold, 0, 0, visited, rows, cols);
        return rtn;
    }
    void helper(int& rtn, int threshold, int i, int j, 
                vector<vector<bool> >& visited, int rows, int cols) {
        if (i < 0 || j < 0 || i >= rows || j >= cols || visited[i][j] ||
            sum_(i, j) > threshold) return;
        rtn++;
        visited[i][j] = true;
        helper(rtn, threshold, i+1, j, visited, rows, cols);
        helper(rtn, threshold, i-1, j, visited, rows, cols);
        helper(rtn, threshold, i, j+1, visited, rows, cols);
        helper(rtn, threshold, i, j-1, visited, rows, cols);
    }
};
// 运行时间：3ms
// 占用内存：616k
