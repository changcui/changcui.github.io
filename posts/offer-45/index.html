扑克牌顺子

Posted at — Mar 11, 2020
// 思路一：失败的条件是有非零重复值，或者最大值减去非零最小值大于4
class Solution {
public:
    bool IsContinuous(vector<int> numbers) {
        if (numbers.size() == 0) return false;
        int min_ = 14, max_ = -1;
        unordered_set<int> s;
        for (auto i : numbers) {
            if (i && s.count(i)) return false;
            s.insert(i);
            if (i) min_ = min(min_, i);
            max_ = max(max_, i);
        }
        return max_ - min_ < 5;
    }
};
// 运行时间：2ms
// 占用内存：504k
