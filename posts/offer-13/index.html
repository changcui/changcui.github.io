调整数组顺序使奇数位于偶数前面

Posted at — Feb 5, 2020
// 思路一：空间换时间，O(n)
class Solution {
public:
    void reOrderArray(vector<int> &array) {
        vector<int> tmp;
        for (auto i : array) if (i & 1) tmp.push_back(i);
        for (auto i : array) if (!(i & 1)) tmp.push_back(i);
        array = tmp;
    }
};
// 运行时间：4ms
// 占用内存：504k

// 思路二：冒泡排序，但是内循环必须从后往前
class Solution {
public:
    void reOrderArray(vector<int> &array) {
        for (int i = 0; i < array.size(); i++) {
            for (int j = array.size() - 1; j > i; j--) {
                if (!(array[j-1] & 1) && (array[j] & 1)) 
                    swap(array[j-1], array[j]);
            }
        }
    }
};
// 运行时间：4ms
// 占用内存：376k

// 思路三：插入排序，维护一个排好的奇数个数变量
class Solution {
public:
    void reOrderArray(vector<int> &array) {
        int cnt = 0;
        for (int i = 0; i < array.size(); i++) {
            if (array[i] & 1) {
                int j = i;
                while (j > cnt) {swap(array[j], array[j-1]); j--;}
                cnt++;
            }
        }
    }
};
// 运行时间：4ms
// 占用内存：480k
