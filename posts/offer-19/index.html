顺时针打印矩阵

Posted at — Mar 7, 2020
// 思路一：确定四个角的位置，然后按照右，下，左，上的顺序，需要注意的是左和上的时候，需要判断是否有新的一行或一列，不然会输出相同的元素
class Solution {
public:
    vector<int> printMatrix(vector<vector<int> > matrix) {
        vector<int> result;
        if (matrix.empty() || matrix[0].empty()) return result;
        int m = matrix.size(), n = matrix[0].size();
        int l = 0, r = n - 1, t = 0, d = m - 1;
        while (l <= r && t <= d) {
            for (int i = l; i <= r; i++) result.push_back(matrix[t][i]);
            for (int i = t + 1; i <= d; i++) result.push_back(matrix[i][r]);
            for (int i = r - 1; i >= l && d > t; i--) 
                result.push_back(matrix[d][i]);
            for (int i = d - 1; i > t && r > l; i--) 
                result.push_back(matrix[i][l]);
            l++, r--, t++, d--;
        }
        return result;
    }
};
// 运行时间：4ms
// 占用内存：464k
