202. Happy Number

Posted at — Sep 5, 2019
class Solution {
public:
    int helper(int n) {
        int rtn = 0;
        while (n) {
            auto tmp = n % 10;
            rtn += tmp * tmp;
            n = (n - tmp) / 10;
        }
        return rtn;
    }
    bool isHappy(int n) {
        auto tmp = helper(n);
        set<int> nums;
        nums.insert(n);
        while (1) {
            if (tmp == 1)
                return true;
            if (nums.find(tmp) != nums.end())
                return false;
            nums.insert(tmp);
            tmp = helper(tmp);
        }
        return false;
    }
};
