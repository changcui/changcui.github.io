1122. Relative Sort Array

Posted at — Aug 29, 2019
class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        vector<bool> first(1001, false);
        for (auto i : arr2) {
            first[i] = true;
        }
        auto len = arr1.size();
        for (int i = 0; i < len;) {
            auto rt = find(arr2.begin(), arr2.end(), arr1[i]);
            if (rt != arr2.end()) {
                if (first[arr1[i]]) {
                    first[arr1[i]] = false;
                } else {
                    arr2.insert(rt, arr1[i]);
                }
                arr1.erase(arr1.begin() + i);
                len -= 1;
            } else {
                i += 1;
            }
        }
        sort(arr1.begin(), arr1.end());
        for (auto i : arr1) {
            arr2.emplace_back(i);
        }
        return arr2;
    }
};
