1170. Compare Strings by Frequency of the Smallest Character

August 30, 2019

class Solution {
public:
    int f(string s) {
        auto rtn = 1;
        auto cur = s[0];
        for (int i = 1; i < s.size(); i++) {
            if (s[i] < cur) {
                rtn = 1;
                cur = s[i];
            } else if (s[i] == cur) {
                rtn += 1;
            }
        }
        return rtn;
    }
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> wa(11, 0);
        for (auto w : words)
            wa[f(w)] += 1;
        vector<int> rtn;
        for (auto q : queries) {
            auto tmp = 0;
            for (int i = f(q) + 1; i <= 10; i++)
                tmp += wa[i];
            rtn.push_back(tmp);
        }
        return rtn;
    }
};
