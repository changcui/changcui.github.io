455. Assign Cookies

August 31, 2019

class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        if (g.empty() || s.empty())
            return 0;
        sort(g.begin(), g.end());
        sort(s.begin(), s.end());
        int rtn = 0;
        for (auto i = s.back();;) {
            auto j = g.back();
            while (i < j) {
                g.pop_back();
                if (g.empty())
                    break;
                j = g.back();
            }
            if (!g.empty()) {
                rtn += 1;
                s.pop_back();
                g.pop_back();
                if (s.empty() || g.empty())
                    break;
                i = s.back();
            } else {
                break;
            }
        }
        return rtn;
    }
};
