从尾到头打印链表

Posted at — Feb 1, 2020
// 思路一：从头到尾储存，然后使用 reverse 反向
class Solution {
public:
    vector<int> printListFromTailToHead(ListNode* head) {
        vector<int> result;
        while (head) {result.push_back(head->val); head = head -> next;}
        reverse(result.begin(), result.end());
        return result;
    }
};
// 运行时间：3ms
// 占用内存：608k

// 思路二：递归调用，出口设置在链表尾，和使用栈是一个效果
class Solution {
public:
    vector<int> printListFromTailToHead(ListNode* head) {
        vector<int> result;
        helper(head, result);
        return result;
    }
    void helper(ListNode* head, vector<int>& result) {
        if (!head) return;
        helper(head->next, result); 
        result.push_back(head->val);
    }
};
// 运行时间：2ms
// 占用内存：460k
