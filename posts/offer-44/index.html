翻转单词顺序列

Posted at — Mar 11, 2020
// 思路一：先翻转字符串，再按照 word 翻转
class Solution {
public:
    string ReverseSentence(string str) {
        if (str.size() == 0) return str;
        reverse(str.begin(), str.end());
        for (int i = 0; i < str.size();) {
            int end = i;
            while (end < str.size() && str[end] != ' ') end++;
            reverse(str.begin()+i, str.begin()+end);
            while (end < str.size() && str[end] == ' ') end++;
            if (end < str.size()) i = end;
            else break；
        }
        return str;
    }
};
// 运行时间：4ms
// 占用内存：592k
