链表中倒数第k个结点

Posted at — Feb 5, 2020
// 思路一：用数组直接存起来，空间开销O(n)，时间开销O(n)
class Solution {
public:
    ListNode* FindKthToTail(ListNode* pListHead, unsigned int k) {
        vector<ListNode*> tmp;
        auto p = pListHead;
        while (p) {tmp.push_back(p); p = p -> next;}
        if (tmp.size() < k) return NULL;
        return tmp[tmp.size() - k];
    }
};
// 运行时间：4ms
// 占用内存：532k

// 思路二：双指针，指针一快指针二 k-1 步，当指针二为空时，指针一为倒数第K个
class Solution {
public:
    ListNode* FindKthToTail(ListNode* pListHead, unsigned int k) {
        auto p = pListHead, q = pListHead;
        int cnt = 0, nowk = k;
        while (p) {
            cnt++;
            if (nowk-- < 1) q = q -> next;
            p = p -> next;
        }
        if (cnt < k) return NULL;
        return q;
    }
};
// 运行时间：5ms
// 占用内存：504k
