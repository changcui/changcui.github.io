1078. Occurrences After Bigram

August 30, 2019

c++ does not have split

class Solution(object):
    def findOcurrences(self, text, first, second):
        """
        :type text: str
        :type first: str
        :type second: str
        :rtype: List[str]
        """
        words = text.split(' ')
        rtn = []
        for i in range(len(words)-2):
            tmp = words[i:i+3]
            if first == tmp[0] and second == tmp[1]:
                rtn.append(tmp[2])
        return rtn
